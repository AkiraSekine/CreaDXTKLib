使用例

                    #include "CreaDXTKLib/Scene.h"
                    #include "Draw/Image.h"
                    #include "Math/Vector2.h"
                    #include "Math/Input.h"

                    using namespace CreaDXTKLib;
                    using namespace CreaDXTKLib::Draw;
                    using namespace CreaDXTKLib::Math;
                    using namespace CreaDXTKLib::Input;

                    // シーンクラスを基底としたメインシーンクラスの定義
                    class InputScene : public Scene
                    {
                    public:

                        // コンストラクタに開始処理を記述
                        InputScene()
                        {
                            // 入力設定の追加
                            Inputs::Instance().Add(L"Right", Keys::D, 1.0f, CheckMode::Press);
                            Inputs::Instance().Add(L"Right", Keys::A, -1.0f, CheckMode::Press);

                            // InputSceneで使用する画像を読み込む。
                            Image::Instance().Load(L"image.png", L"Test");
                        }

                        // デストラクタに終了処理を記述
                        ~InputScene()
                        {
                            // 入力設定の破棄
                            Inputs::Instance().Erase(L"Right");
                            
                            // InputSceneが終了するので画像を破棄
                            Image::Instance().Erase(L"Test");
                        }

                        // 更新処理
                        virtual void Update(float _elapsedTime) override
                        {
                            // 入力を取得
                            float input = Inputs::Instance().GetValue(L"Right");
                            
                            // Aが押されていたら左へ、Dが押されていたら右へ画像を移動
                            m_imagePos += Vector2(100.0f, 0.0f) * _elapsedTime * input;
                        }

                        // 描画処理
                        virtual void OnRender() override
                        {
                            //画像を描画
                            Image::Instance().Draw(L"Test", m_imagePos);
                        }
                        
                    private:

                        // 画像の座標
                        Vector2 m_imagePos = Vector2::zero;
                    }
