使用例

                    #include "CreaDXTKLib/Scene.h"
                    #include "CreaDXTKLib/Object.h"
                    #include "Draw/Image.h"
                    #include "Math/Vector2.h"

                    using namespace CreaDXTKLib;
                    using namespace CreaDXTKLib::Draw;
                    using namespace CreaDXTKLib::Math;

                    // オブジェクトの定義
                    class Player : public Object
                    {
                        OBJECT2D(Player)

                    public:

                        virtual void Update(float _elapsedTime) override
                        {
                            // オブジェクトの移動
                            Position(Position() + Vector2(100.0f, 0.0f) * _elapsedTime);
                        }
                    }

                    // シーンクラスを基底としたメインシーンクラスの定義
                    class TransformScene : public Scene
                    {
                    public:

                        // コンストラクタに開始処理を記述
                        TransformScene()
                        {
                            // TransformSceneで使用する画像を読み込む。
                            Image::Instance().Load(L"image.png", L"Test");

                            // プレイヤーの初期位置をTransform2Dで指定
                            Transform2D playerTransform;
                            playerTransform = Transform2D(Vector2::zero, 0.0f, Vector2::one);

                            // プレイヤーのインスタンスを生成
                            m_player = Player(L"Test", playerTransform);
                        }

                        // デストラクタに終了処理を記述
                        ~TransformScene()
                        {
                            // TransformSceneが終了するので画像を破棄する。
                            Image::Instance().Erase(L"Test");
                        }

                        // 更新処理
                        virtual void Update(float _elapsedTime) override
                        {
                            // オブジェクトの更新処理
                            m_player.Update(_elapsedTime);
                        }

                        // 描画処理
                        virtual void OnRender() override
                        {
                            // オブジェクトを描画
                            m_Player.Draw();
                        }
                        
                    private:

                        // Vector2を画像の座標として使用
                        Vector2 m_imagePos = Vector2::zero;

                        // オブジェクト
                        Player m_player;
                    }
