使用例

                    #include "CreaDXTKLib/Scene.h"
                    #include "Audio/Sound.h"

                    using namespace CreaDXTKLib;
                    using namespace CreaDXTKLib::Audio;

                    // シーンクラスを基底としたシーンクラスの定義
                    class SoundScene : public Scene
                    {
                    public:

                        // コンストラクタに開始処理を記述
                        SoundScene()
                        {
                            Sound::Instance().Load(L"Test", L"sound.wav");

                            Sound::Instance().Play(L"Test");
                        }

                        // デストラクタに終了処理を記述
                        ~SoundScene()
                        {
                            Sound::Instance().Destroy(L"Test");
                        }

                        // 更新処理
                        virtual void Update(float _elapsedTime) override
                        {
                        }

                        // 描画処理
                        virtual void OnRender() override
                        {
                        }
                    }
