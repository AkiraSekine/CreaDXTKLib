メソッド名 概要 返り値 引数 備考

Log デバッグテキストの出力 void - std::wstring _str 出力文字列(フォーマット) デバッグ時のみ使用可能

使用例

                    #include "CreaDXTKLib/Scene.h"
                    #include "Utility/Debug.h"

                    using namespace CreaDXTKLib;
                    using namespace CreaDXTKLib::Utility;

                    // シーンクラスを基底としたメインシーンクラスの定義
                    class DebugScene : public Scene
                    {
                    public:

                        // 更新処理
                        virtual void Update(float _elapsedTime) override
                        {
                            // 前フレームからの経過時間を出力
                            Debug::Log(L"%f\n", _elapsedTime);
                        }

                        // 描画処理
                        virtual void OnRender() override
                        {
                        }
                    }
